sudo apt update 
sudo apt install openjdk-17-jdk #tomcat 11 need java 17 to be installed
cd /opt/
sudo wget https://dlcdn.apache.org/tomcat/tomcat-11/v11.0.18/bin/apache-tomcat-11.0.18.tar.gz #download the tomcat version 11
sudo tar -xzvf apache-tomcat-11.0.18.tar.gz #extract the downloaded file
sudo /opt/apache-tomcat-11.0.18/bin/startup.sh #start tomcat
sudo ss -antup #to check that tomcat is running
sudo vim /opt/apache-tomcat-11.0.18/conf/server.xml #to change the tomcat port to 7070
sudo /opt/apache-tomcat-11.0.18/bin/shutdown.sh
sudo /opt/apache-tomcat-11.0.18/bin/startup.sh #to restart the tomcat after changing the port
sudo ss -antup #to check that it's now listening on port 7070 or by accessing http://Server-IP:7070
sudo apt install nginx
sudo systemctl enable nginx #to start automatically at boot
sudo systemctl start nginx #to start the service now
sudo vim /etc/nginx/sites-available/tomcat #to write the configuration file
sudo ln -s /etc/nginx/sites-available/tomcat /etc/nginx/sites-enabled/ #to link the new configuration file and enable the site
sudo rm /etc/nginx/sites-enabled/default #remove default site
sudo systemctl restart nginx
sudo /etc/hosts #to add this entry 192.168.182.128 mytomcat.example.com in order to be able to access it from my browser
